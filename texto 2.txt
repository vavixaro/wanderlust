function renderDayNav(r) {
  const nav = document.getElementById('day-nav');
  const days = ROUTES[r].days;
  nav.innerHTML = '';
  days.forEach((d, i) => {
    const btn = document.createElement('button');
    btn.className = 'day-btn' + (i===0?' active':'');
    btn.textContent = d.btn;
    btn.onclick = () => {
      document.querySelectorAll('#day-nav .day-btn').forEach(b=>b.classList.remove('active'));
      btn.classList.add('active');
      renderDay(r, i);
    };
    nav.appendChild(btn);
  });
}

function doSearch(){

  document.getElementById('loading').classList.add('on');

  const msgs=['🔗 Conectando con API...','🔍 Buscando mejores ofertas...','📊 Comparando precios...','✅ Resultados listos'];

function showRes(){
  document.getElementById('loading').classList.remove('on');
  document.getElementById('r-section').classList.add('on');
  if(searchMode==='flight'){
    const from=document.getElementById('f-from').value.split(' ')[0];
    const to=document.getElementById('f-to').value.split(' ')[0];
    document.getElementById('r-title').textContent=`Vuelos ${from} → ${to}`;
    document.getElementById('r-count').textContent=`${FLIGHTS.length} vuelos encontrados · ${paxCount} pasajero${paxCount>1?'s':''} · Precio por persona`;
    renderFlights(FLIGHTS);

    const sLabel=f.stops===0?'<span class="sbadge sb-d">Directo</span>':f.stops===1?`<span class="sbadge sb-1">1 escala · ${f.stop}</span>`:`<span class="sbadge sb-2">2 escalas</span>`;
    c.innerHTML+=`<div class="f-card ${isBest?'best':''}" style="animation-delay:${i*.08}s">
      ${isBest?'<div class="best-tag">🏷 Mejor precio</div>':''}
      <div>
        <div class="f-airline-row">${f.airline} · ${f.code}</div>
        <div class="f-main">
          <div><div class="f-time">${f.dep}</div><div class="f-ap">${f.from}</div></div>
          <div class="f-mid">
            <div class="f-dur">${f.dur}</div>
            <div class="f-line"><div class="f-ln"></div><span class="f-pl">✈</span><div class="f-ln"></div></div>
            <div class="f-stops-row" style="margin-top:5px">${sLabel}</div>

        </div>
        <div style="margin-top:10px;display:flex;gap:14px;font-size:11px;color:var(--muted)">

  });
}

// ── SORT ──
function doSort(by,btn){
  document.querySelectorAll('.sort-btn').forEach(b=>b.classList.remove('active'));
  btn.classList.add('active');
  if(searchMode==='flight'){
    const s=[...FLIGHTS].sort((a,b)=>by==='price'?a.price-b.price:by==='stops'?a.stops-b.stops:a.dur.localeCompare(b.dur));
    renderFlights(s);
  } else {
    const cin=document.getElementById('h-in').value,cout=document.getElementById('h-out').value;
    const nights=cin&&cout?Math.round((new Date(cout)-new Date(cin))/86400000):3;
    const s=[...HOTELS].sort((a,b)=>by==='price'?a.price-b.price:b.rating-a.rating);
    renderHotels(s,nights);
  }
}

function refreshBgt(){
  const pct=Math.min(100,Math.round(totalSpent/totalBudget*100));
  const rem=totalBudget-totalSpent;
  document.getElementById('bgt-spent').textContent=`€${totalSpent.toLocaleString()}`;
  document.getElementById('bgt-total-lbl').textContent=`€${totalBudget.toLocaleString()}`;
  document.getElementById('bgt-remaining').textContent=`€${rem.toLocaleString()} restantes`;
  document.getElementById('bgt-pct').textContent=`${pct}%`;
  const bar=document.getElementById('main-prog');
  bar.style.width=pct+'%';
  bar.className='prog-fill'+(pct>90?' over':pct<60?' good':'');
}
function addExpItem(name,amt,icon){
  totalSpent+=amt;
  const list=document.getElementById('exp-list');
  const d=document.createElement('div');
  d.className='exp-item';d.style.animation='fadeUp .3s ease both';
  d.innerHTML=`<div class="exp-l"><span class="exp-ico">${icon}</span><div><div class="exp-nm">${name}</div><div class="exp-ct">Añadido ahora</div></div></div><div class="exp-amt amt-neg">−€${amt}</div>`;
  list.prepend(d);
  refreshBgt();
}

}

function speakPhrase(text,lang){
  if(!window.speechSynthesis){toast('⚠️ Audio no disponible');return;}
  const u=new SpeechSynthesisUtterance(text);
  u.lang=lang;u.rate=0.8;
  window.speechSynthesis.cancel();
  window.speechSynthesis.speak(u);
}

function doTranslate(){
  const input=document.getElementById('tr-input')?.value.trim();
  if(!input) return;
  const out=document.getElementById('tr-output');
  out.innerHTML='<span style="color:var(--muted2);font-size:13px;">⏳ Traduciendo...</span>';
  const url=`https://translate.googleapis.com/translate_a/single?client=gtx&sl=es&tl=ja&dt=t&dt=rm&q=${encodeURIComponent(input)}`;
  fetch(url).then(r=>r.json()).then(data=>{
    const t=data[0]?.map(s=>s[0]).join('')||'—';
    out.innerHTML=`<span style="font-size:16px;line-height:1.8;">${t}</span>`;
    const rm=data[0]?.map(s=>s?.[3]||'').filter(Boolean).join(' ');
    const pb=document.getElementById('tr-phonetic-box');
    if(rm&&pb){document.getElementById('tr-phonetic').textContent=rm;pb.style.display='block';}
  }).catch(()=>{out.innerHTML='<span style="color:var(--muted2);">Sin conexión — usa las frases esenciales</span>';});
}
